layout(location=0) out vec3 v_col;

void main() {
    float acc = 0.0;
    for (int i = 0; (i < 8); i++) {
        if ((i == 3)) {
            continue;
        }
        if ((i > 6)) {
            break;
        }
        acc += float(i);
    }
    float s = 0.0;
    int k = 4;
    do {
        s += float(k);
        k--;
    } while ((k > 0));
    vec4 v = vec4(0.0, 0.0, 0.0, 1.0);
    v.xy = vec2((s * 0.01), (acc * 0.01));
    const vec2 pos[3] = vec2[3](vec2(-0.5, -0.5), vec2(0.5, -0.5), vec2(0.0, 0.5));
    gl_Position = vec4(pos[gl_VertexID], 0.0, 1.0);
    v_col = lerp3(vec3(1.0, 0.0, 0.0), vec3(0.0, 1.0, 0.0), clamp01((float(gl_VertexID) * 0.5)));
}

float clamp01(float x) {

    return (a + ((b - a) * t));
}

layout(location=0) in vec3 v_col;
layout(location=0) out vec4 f_color;

void main() {
    int m = int((v_col.x * 8.0));
    float r = 0.0;
    if (m == 1 || m == 2) {
        r = 0.25;
    }
    else if (m == 3) {
        r = 0.5;
        r = (r + 0.25);
    }
    else if (m == 4) {
        r = (r + 0.25);
    }
    else {
        r = 1.0;
    }
    float a = clamp01(v_col.x);
    if ((a < 0.01)) {
        discard;
    }
    f_color = vec4((v_col * r), a);
    return;
}
