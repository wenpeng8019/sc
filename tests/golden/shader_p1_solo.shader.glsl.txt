layout(set=0, binding=0) uniform sampler2D tex2d;

layout(set=0, binding=3) uniform sampler2DArray texarr;

layout(location=0) in vec2 v_uv2;
layout(location=1) in vec3 v_uv3;
layout(location=0) out vec4 f_color;

void main() {
    const vec4 c2d = texture(tex2d, v_uv2);
    const vec4 c2d_lod = textureLod(tex2d, v_uv2, 0.0);
    const vec4 c3d = texture(tex3d, v_uv3);
    const vec4 c3d_lod = textureLod(tex3d, v_uv3, 1.0);
    const vec4 ccube = texture(texcube, v_uv3);
    const vec4 ccube_lod = textureLod(texcube, v_uv3, 1.5);

    const vec4 carr_lod = textureLod(texarr, v_uv3, 0.5);

    return;
}
