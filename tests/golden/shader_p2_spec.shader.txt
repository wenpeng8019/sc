#include <metal_stdlib>
#include <simd/simd.h>

using namespace metal;

struct XBuf_blk
{
    float x[1];
};

constant uint TILE_tmp [[function_constant(0)]];
constant uint TILE = is_function_constant_defined(TILE_tmp) ? TILE_tmp : 64u;
constant float GAIN_tmp [[function_constant(1)]];
constant float GAIN = is_function_constant_defined(GAIN_tmp) ? GAIN_tmp : 1.5;

kernel void cs_spec(device XBuf_blk& XBuf [[buffer(0)]], uint3 gl_GlobalInvocationID [[thread_position_in_grid]])
{
    uint TILE_1 = TILE;
    float GAIN_1 = GAIN;
    float PLAIN = 2.0;
    if (gl_GlobalInvocationID.x < TILE_1)
    {
        XBuf.x[gl_GlobalInvocationID.x] = (XBuf.x[gl_GlobalInvocationID.x] * GAIN_1) * PLAIN;
    }
}
