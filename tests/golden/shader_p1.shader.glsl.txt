void main() {
    const vec4 c2d = texture(Tex2DBlock.tex, v_uv0);
    const vec4 c2d_lod = textureLod(Tex2DBlock.tex, v_uv0, 0.0);
    const vec4 c3d = texture(Tex2DBlock.tex3d, v_uv3);

}
