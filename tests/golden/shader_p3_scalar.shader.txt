// ===== cs_f16.metal20300.metal =====
#include <metal_stdlib>
#include <simd/simd.h>

using namespace metal;

struct Params_blk
{
    uint n;
};

struct WBuf_blk
{
    half w[1];
};

struct OBuf_blk
{
    long acc;
    float o[1];
};

kernel void cs_f16(constant Params_blk& Params [[buffer(0)]], device WBuf_blk& WBuf [[buffer(1)]], device OBuf_blk& OBuf [[buffer(2)]], uint3 gl_GlobalInvocationID [[thread_position_in_grid]])
{
    half HALF_SCALE = half(0.5);
    long BIG = 1099511627776l;
    if (gl_GlobalInvocationID.x < Params.n)
    {
        half h = WBuf.w[gl_GlobalInvocationID.x];
        h = (h * HALF_SCALE) + half(0.25);
        OBuf.o[gl_GlobalInvocationID.x] = float(h);
        if (gl_GlobalInvocationID.x == uint(0))
        {
            OBuf.acc = BIG + long(ulong(gl_GlobalInvocationID.x));
        }
    }
}
