#version 450
// generated from vs_main (vert) target vulkan450

layout(set=0, binding=1) uniform sampler2D tex;
layout(set=0, binding=0, std140) uniform Camera_blk {
    mat4 mvp;
} Camera;

vec3 tm_aces(vec3 c) {
    return ((c * ((c * 2.51) + 0.03)) / ((c * ((c * 2.43) + 0.59)) + 0.14));
}

vec3 tm_reinhard(vec3 c) {
    return (c / (c + vec3(1.0, 1.0, 1.0)));
}

layout(location=0) in vec3 pos;
layout(location=1) in vec2 uv;
layout(location=0) out vec2 v_uv;

void main() {
    gl_Position = (Camera.mvp * vec4(pos, 1.0));
    v_uv = uv;
}
